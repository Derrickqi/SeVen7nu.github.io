def numSort(arr):
    n = len(arr)

    # 遍历所有数组元素
    for i in range(n):
        
        # 将列表元素俩俩依次对比并交换顺序
        for j in range(0,n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]

arr = [64, 34, 25, 12, 22, 11, 90]

if __name__ == '__main__':
    numSort(arr)
    
    # 输出排列后的数组
    for i in range(len(arr)):
        print("%d" %arr[i],end=" ")
